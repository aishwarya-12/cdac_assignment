3. write a procedure to display empno,ename,deptno,dname for all employees with sal > given salary. 
Pass salary as a parameter to procedure.

delimiter //
create procedure check_sal(in salary float(7,2))
begin
	declare v int default 0;
	declare vempno, vdeptno int;
	declare vename, vdname varchar(25);
	declare vsal float(7,2);
	declare empcur cursor for select empno,ename,sal,d.deptno,dname from emp e inner join dept d on e.deptno = d.deptno where sal > salary;
	declare continue handler for NOT FOUND set v=1;
	open empcur;
	label1 : loop
		fetch empcur into vempno, vename,vsal,vdeptno,vdname;
		if v=1 then
			leave label1;
		end if;
		select vempno, vename,vsal,vdeptno,vdname;
	end loop;
	close empcur;
end//

4. write a procedure to find min,max,avg of salary and number of employees in the
given deptno.
deptno --→ in parameter
min,max,avg and count ---→ out type parameter execute procedure and then display values min,max,avg and count.

delimiter //
create procedure getdata1(in dno int, out minsal double, out maxsal double, out avgsal double, out countemp int)
begin
	select max(sal), min(sal), avg(sal), count(*) into maxsal, minsal, avgsal, countemp
	from emp
	where deptno = dno;
	select minsal, maxsal, avgsal, countemp;
end//

5. write a procedure to display all pid,pname,cid,cname and salesman name(use product,category and salesman table)

delimiter //
create procedure displayprod(out ppid int, out ccid int, out name varchar(20), out salesname varchar(20))
begin
	declare v int default 0;
	declare salecur cursor for select p.pid, c.cid, c.cname, s.sname 
		from product p inner join category c on p.cid = c.cid 
		inner join salesman s on p.sid = s.sid;
	declare continue handler for NOT FOUND set v = 1;
	open salecur;
	label1: loop
		fetch salecur into ppid, ccid, name, salesname;
		if v=1 then
			leave label1;
		end if;
		select ppid, ccid, name, salesname;
	end loop;
	close salecur;
end//

create procedure displayEmp1(in dno1 int)
begin
	declare v, cnt int default 0;
	declare eno, dno int;
	declare name, ejob, status varchar(20);
	declare salary,avgsal double;
	declare empcur cursor for select empno, ename, job, sal, deptno from emp where deptno = dno1;
	declare continue handler for NOT FOUND set v = 1;
	open empcur;
	label : loop
		fetch empcur into eno, name, ejob, salary, dno;
                if v=1 then
			if cnt = 0 then
    			select "Emp table is empty" errmsg;
    
			end if;
			leave label;
		end if;
		set cnt = cnt +1;

		select avg(sal) into avgsal from emp
		where deptno = dno1;

8. Write a procedure to update salary in emp table based on following rules.
Exp< =35 then no Update
Exp> 35 and <=38 then 20% of salary
Exp> 38 then 25% of salary

delimiter //
create procedure updatesal1()
begin
	declare v int default 0;
	declare name varchar(20);
	declare eno int;
	declare ehiredate date;
	declare salary double;
	declare empcur cursor for select empno, ename, hiredate, sal from emp;
	declare continue handler for NOT FOUND set v =1;
	open empcur;
	label1 : loop
		fetch empcur into eno,name,ehiredate, salary;
		if v = 1 then
			leave label1;
		end if;
		if (timestampdiff(YEAR, ehiredate, curdate())) > 35 and (timestampdiff(YEAR, ehiredate, curdate())) <= 38 then
			update emp
			set sal = salary + (salary*0.2)
			where empno = eno;  
		elseif timestampdiff(YEAR, ehiredate, curdate()) > 38  then
			update emp
			set sal = salary + (salary*0.25)
			where empno = eno;  
		end if;
		select eno,name,ehiredate, salary;
	end loop;
	close empcur;
end//
